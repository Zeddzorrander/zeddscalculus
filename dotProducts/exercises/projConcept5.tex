\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
      \begin{axis}[
          xmin=-5,xmax=5,ymin=-5,ymax=5,
          clip=false,
          unit vector ratio*=1 1 1,
          %axis lines=none,
          grid = major,
          ytick={-5,-4,...,5},
	  xtick={-5,-4,...,5},
          ticks = none,
        ]
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (1,-3)};
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (1.11,3.69)};
          \node[left] at (axis cs:.5, -1.5) [penColor2] {$\vec{w}$};
          \node[right] at (axis cs:.56, 1.845) [penColor] {$\vec{v}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Assuming that the components of $\proj_{\vec{w}}(\vec{v})$ are whole
  numbers,
  \[
  \proj_{\vec{w}}(\vec{v}) = \vector{\answer{-1},\answer{3}}
  \]
  You should be able to answer this question \textbf{without} any
  computation.  
\end{exercise}
