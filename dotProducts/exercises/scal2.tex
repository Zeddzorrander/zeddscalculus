\begin{exercise}
Let $\vec{u}=\vector{1,3}$ and $\vec{v}=\vector{5,5}$. Find the
scalar projection of $\vec{v}$ along $\vec{u}$.
\begin{prompt}
\[
\scal_\vec{u}(\vec{v})=\answer{2\sqrt{10}}
\]
\end{prompt}
