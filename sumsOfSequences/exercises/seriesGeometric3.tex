\begin{exercise}
Consider the geometric series:

\[
\sum_{k=0}^{\infty} \frac{2^{2k+1}}{3^k}
\]

Determine whether the series converges or diverges.  If it converges, give the value to which it converges.

\begin{multipleChoice}
\choice[correct]{The series diverges.}
\choice{The series converges.}
\end{multipleChoice}

\end{exercise}

\begin{exercise}
Consider the geometric series:

\[
\sum_{k=0}^{\infty} 2^{3-2k}
\]

Determine whether the series converges or diverges.  If it converges, give the value to which it converges.

\begin{multipleChoice}
\choice{The series diverges.}
\choice[correct]{The series converges.}
\end{multipleChoice}

\begin{exercise}
The series converges to $\answer{\frac{32}{3}}$.
\end{exercise}
\end{exercise}
