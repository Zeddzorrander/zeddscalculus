\begin{exercise}
For which values of $a$ does the geometric series below converge? 
\[
\sum_{k=0}^{\infty} \left(\frac{1}{2}\right)^{ak}
\]

The series:

\begin{multipleChoice}
\choice{converges for all values of $a$}
\choice[correct]{converges if $a >0$.}
\choice{converges if $a \geq 0$.}
\choice{converges if $|a| <1$.}
\choice{converges if $|a| \leq1$.}
\choice{diverges for all values of $a$.}
\choice{none of these}
\end{multipleChoice}

\begin{hint}
The geometric series $\sum_{k=k_0}^{\infty} cr^k$ converges if $|r| < \answer{1}$.  Here, $r = \answer{\left(\frac{1}{2}\right)^a}$.
\end{hint}

\end{exercise}
