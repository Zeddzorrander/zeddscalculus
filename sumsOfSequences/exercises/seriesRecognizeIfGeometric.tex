\begin{exercise}
Select all of the following series that are geometric:

\begin{selectAll}
\choice[correct]{$\sum_{k=6}^{\infty} \left(\frac{3}{2}\right)^k$}
\choice[correct]{$\sum_{k=5}^{\infty} \frac{2^{3k-1}}{5^5}$}
\choice{$\sum_{k=0}^{\infty} \frac{2k+3^k}{4^k}$}
\choice[correct]{$\sum_{k=2}^{\infty} 4^{1-2k}$}
\choice{$\sum_{k=1}^{\infty} \frac{k}{2^k}$}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{5}{6^k}$}
\end{selectAll}

\end{exercise}
