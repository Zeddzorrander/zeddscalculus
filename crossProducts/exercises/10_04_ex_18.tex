\begin{exercise}
Let $\vec{u}$ and $\vec{v}$ be vectors in $\mathbb{R}^3$. Suppose
$|\vec{u}|=3$, $|\vec{v}|=4$ and that the angle between $\vec{u}$ and
$\vec{v}$ is $\theta=\pi$. Compute $|\vec{u}\cross\vec{v}|$.

\begin{prompt}
\[
|\vec{u}\cross\vec{v}|=\answer{0}
\]
\end{prompt}


\end{exercise}
