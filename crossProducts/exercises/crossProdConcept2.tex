\outcome{Compute dot products.}

\begin{document}
\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
        \begin{axis}[
            unit vector ratio*=1 1 1,
            axis lines=none
          ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (0,-2)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (4,1)};
          \node[left] at (axis cs:0, -1) [penColor] {$\vec{v}$};
          \node[above] at (axis cs:2, .5) [penColor2] {$\vec{w}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Does $\vec{v}\cross\vec{w}$ point toward you, or away from you, or
  neither?
  \begin{multipleChoice}
    \choice[correct]{toward me}
    \choice{away from me}
    \choice{neither}
  \end{multipleChoice}
  \end{exercise}
