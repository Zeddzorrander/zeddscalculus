\begin{exercise}
Suppose that $\{a_n\}_{n=1}$ is a sequence of positive terms and it is known that $\sum_{k=1}^{n} a_k = \frac{2n^2+4}{n^2+\ln^3(n)}$.  Then, which of the following must be true?

\begin{selectAll}
\choice{$\sum_{k=1}^{\infty} a_k$ diverges by the divergence test.}
\choice{$\lim_{n \to \infty} a_n = 2$.}
\choice[correct]{$\sum_{k=1}^{\infty} a_k$ converges.}
\choice[correct]{$\sum_{k=1}^{\infty} a_k$ converges to $2$.}
\choice[correct]{Remainders can be defined for $\sum_{k=1}^{\infty} a_k$.}
\choice{Remainders can be defined for $\sum_{k=1}^{\infty} s_k$.}
\end{selectAll}

\begin{hint}
Make sure that you process and classify the information that you are given;  the given information $\sum_{k=1}^{n} a_k = \frac{2n^2+4}{n^2+\ln^3(n)}$ tells us something immediately about:

\begin{multipleChoice}
\choice{the sequence $\{a_n\}$.}
\choice[correct]{the sequence of partial sums for $\{a_n\}$.}
\end{multipleChoice}
\end{hint}
\end{exercise}
