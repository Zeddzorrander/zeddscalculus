\begin{exercise}

  Which of the following sequences are geometric sequences for
  $n=1,2,3,\dots$?
  \begin{selectAll}
    \choice{$a_n = -3n + 5$}
    \choice[correct]{$a_n = -5\cdot 7^{n+4}$}
    \choice[correct]{$a_n = \frac{5^n}{2^{2n+1}}$}
    \choice[correct]{$a_n = \left(\frac{1}{3}\right)\cdot a_{n-1}$ where $a_1 = 10$}
    \choice{$a_n = \frac{4n^2+n-14}{n+2}$}
    \choice{$a_n = n\cdot \cos(2\pi n) +4$}
    \choice{$a_n = n \cdot 2^n$}
  \end{selectAll}
  \begin{hint}
   Write out some terms and look for a common ratio between successive elements in the sequence. 
  \end{hint}




\end{exercise}
