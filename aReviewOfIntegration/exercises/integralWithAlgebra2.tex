\begin{exercise}
Compute the following indefinite integral:

\begin{prompt} (Use $C$ for the constant of integration) \end{prompt}

\[
\int (\sqrt{x}+4)^2  \d x 
\begin{prompt}=\answer{\frac{1}{2}x^2 +\frac{16}{3} x^{3/2}+16x +C}\end{prompt}
\]
\end{exercise}
