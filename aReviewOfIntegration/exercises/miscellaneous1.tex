\begin{exercise}

Select ALL of the following responses that are correct:

\begin{selectAll}  
    \choice[correct]{$\int \ln(x) \d x = x \ln(x) -x +C$}  
    \choice{By setting $u=x^2$, $\int e^{x^2} \d x = \int e^u \d u$}  
    \choice{$\int\frac{1}{x^3+1} \d x = \ln|x^3+1|+C$}  
    \choice[correct]{$\int_0^2 9\arcsin(3x) \d x = \int_0^6 3\arcsin(v) \d v$}  
    \choice[correct]{$\int \sec(4x)\tan(4x) \d x = \frac{1}{4} \sec(4x) +C$}  
    \choice{$\int 2x^2 \d x = \frac{2}{3} x^3$}  
  \end{selectAll}

\end{exercise}
