\begin{exercise}
Compute the following indefinite integral:

\begin{prompt} (Use $C$ for the constant of integration) \end{prompt}

\[
\int (2x-1)\sqrt{4x^2-4x+9} \d x 
\begin{prompt}=\answer{\frac{1}{6}(4x^2-4x+9)^{\frac{3}{2}}+C} \end{prompt}
\]
\end{exercise}
