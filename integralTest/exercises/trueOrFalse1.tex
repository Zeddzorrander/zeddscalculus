\begin{exercise}


Select all of the statements that \emph{must} be true.  Note that there may be several statements that are true! 

 \begin{selectAll}
\choice{If $\lim_{n \to \infty} a_n = 0$, then $\sum_{k=1}^{\infty} a_k$ converges.}
\choice[correct]{If $\sum_{k=1}^{\infty} a_k$ converges, then $\lim_{n \to \infty} a_n = 0$.}
\choice[correct]{If $\lim_{n \to \infty} a_n \neq 0$, then $\sum_{k=1}^{\infty} a_k$ diverges.}
\choice{If $\sum_{k=1}^{\infty} a_k$ diverges, then $\lim_{n \to \infty} a_n \neq 0$.}
 \end{selectAll}
 
 \end{exercise}
