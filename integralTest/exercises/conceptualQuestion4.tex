\begin{exercise}

Suppose that $\{a_n\}$ and $\{b_n\}$ are sequences and it is known that:

\[
\sum_{k=1}^{\infty} a_k = 5 \qquad \textrm{ and } \qquad b_n = 3  \textrm{ for all } n \geq 1
\]
Answer the following:

The series $\sum_{k=1}^{\infty} (a_k-1)$:
\begin{multipleChoice}
\choice{converges to $0$.}
\choice{converges to $4$.}
\choice{converges, but more information is needed to determine its value.}
\choice{could converge or diverge; more information is needed.}
\choice[correct]{diverges by divergence test.}
\end{multipleChoice}

\begin{hint}
Since $\sum_{k=1}^{\infty} a_k$ converges, we know that $\lim_{n \to \infty} = \answer{0}$.  Hence, $\lim_{n \to \infty} (a_n-1) = \answer{-1}$.  Thus:

\begin{multipleChoice}
\choice{$\sum_{k=1}^{\infty} (a_k -1)$ converges to $-1$.}
\choice[correct]{$\sum_{k=1}^{\infty} (a_k -1)$ diverges by the divergence test.}
\end{multipleChoice}


The notation is key here; it would be true that $\sum_{k=1}^{\infty} a_k-1 = 4$ since the notation $\sum_{k=1}^{\infty} a_k-1$ is interpreted as $\left(\sum_{k=1}^{\infty} a_k\right)-1$ (sum the terms in the sequence, \emph{then} subtract $1$).
\end{hint}

\begin{exercise}
The series $\sum_{k=1}^{\infty} (a_k + b_k)$:
\begin{multipleChoice}
\choice{converges to $3$.}
\choice{converges to $\frac{1}{2}$.}
\choice{converges, but more information is needed to determine its value.}
\choice{could converge or diverge; more information is needed.}
\choice[correct]{diverges by divergence test.}
\end{multipleChoice}

\begin{hint}
Since $\sum_{k=1}^{\infty} a_k$ converges, we know $\lim_{n \to \infty} a_n = \answer{0}$.  Hence, $\lim_{n \to \infty} (a_n+b_n) = \answer{3}$.  Thus:

\begin{multipleChoice}
\choice{$\sum_{k=1}^{\infty} (a_k +b_k)$ converges to $3$.}
\choice[correct]{$\sum_{k=1}^{\infty} (a_k +b_k)$ diverges by the divergence test.}
\end{multipleChoice}
\end{hint}
\end{exercise}
