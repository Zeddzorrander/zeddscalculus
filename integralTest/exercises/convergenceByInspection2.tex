\begin{exercise}
Consider the series $\sum_{k=1}^{\infty} \frac{2^k}{k^3}$. Select all that apply:
\begin{multipleChoice}
\choice{This is a geometric series.  It converges.}
\choice{This is a geometric series.  It diverges.}
\choice{This is a $p$-series.  It converges.}
\choice{This is a $p$-series.  It diverges.}
\choice[correct]{This series diverges by the divergence test.}
\end{multipleChoice}

\begin{hint}
What does growth rates tell you about $\lim_{n \to \infty} \frac{2^n}{n^3}$?
\end{hint}
\end{exercise}
