\begin{exercise}


Select all of the statements that \emph{must} be true.  Note that there may be several statements that are true! 

 \begin{selectAll}
\choice[correct]{The series $\sum_{k=2}^{\infty} \frac{3k}{2k+1}$ diverges.}
\choice{If $\lim_{n \to \infty} a_n = 0$, then $\sum_{k=1}^{\infty} a_k$ converges.}
\choice{The series $\sum_{k=1}^{\infty} \frac{2}{k^3}$ diverges.}
\choice[correct]{If $\sum_{k=1}^{\infty} a_k$ converges, then $\lim_{n \to \infty} a_n = 0$.}
\choice{Since $\sum_{k=1}^{\infty} \frac{1}{k^2}$ is a convergent $p$-series, $\sum_{k=1}^{\infty} \frac{1}{k^2} = \sum_{k=3}^{\infty} \frac{1}{k^2}$.}
\choice[correct]{If $\sum_{k=1}^{\infty} a_k$ and $\sum_{k=1}^{\infty} b_k$ converge, then $\sum_{k=1}^{\infty} a_k+b_k$ converges.}
 \end{selectAll}
 
 \end{exercise}
