\outcome{Understand the relationship between the sequence of remainders and the convergence of the series.}

\begin{document}

\begin{exercise}

Select all of the following statements that must be true.

\begin{selectAll}
\choice{It is possible to find a sequence $\{a_n\}_{n=1}$ so $\sum_{k=1}^{\infty} a_k$ diverges and $\lim_{n \to \infty} r_n =1$.}
\choice{It is possible to find a sequence $\{a_n\}_{n=1}$ so $\sum_{k=1}^{\infty} a_k$ converges and $\lim_{n \to \infty} r_n =1$.}
\choice[correct]{It is possible to find a sequence $\{a_n\}_{n=1}$ so $\sum_{k=1}^{\infty} a_k$ converges and $\lim_{n \to \infty} r_n =0$.}
\choice{It is possible to find a sequence $\{a_n\}_{n=1}$ so $\sum_{k=1}^{\infty} a_k$ diverges and $\lim_{n \to \infty} r_n =0$.}
\end{selectAll}

\end{exercise}
