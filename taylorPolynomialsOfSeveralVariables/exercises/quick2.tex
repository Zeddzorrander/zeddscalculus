\begin{exercise}
  Compute the degree $2$ Taylor polynomial for
  \[
  F(x,y) = y^2-3xy
  \]
  centered at $(0,0)$.  The degree $2$ Taylor polynomial is:
  \[
  P_2(x,y) =\answer{y^2-3xy}
  \]
\end{exercise}
