\begin{exercise}

Consider the series 
\[
\sum^{\infty}_{n=1} \frac{5^n}{n^n}
\]
We want to determine whether the series converges or diverges. 


Which of the following methods would be the best to determine convergence of the series:

\begin{multipleChoice}
\choice{integral test}
\choice[correct]{root test}
\choice{geometric series}
\choice{telescoping series}
\end{multipleChoice}



\begin{exercise}

We will use the Root Test to test for convergence of the series. 
We need to consider
\[
\sqrt[n]{a_n}
\]

where $a_n=\answer{ \frac{5^n}{n^n}}$. 

Simplifying we have $ \sqrt[n]{\frac{5^n}{n^n}} =\answer{ \frac{5}{n}}$



\begin{exercise}

Now we need to calculate $\lim_{n\to \infty} \sqrt[n]{\frac{5^n}{n^n}} =\answer{ 0 }$


\begin{exercise}

Since $\lim_{n \to \infty} \sqrt[n]{a_n}=$  \wordChoice{\choice[correct]{less than 1}\choice{greater than 1}\choice{equal to 1}},
we see that the series
\[
\sum^{\infty}_{n=1} \frac{5^n}{n^n}
\]
\begin{multipleChoice}
\choice[correct]{converges}
\choice{diverges}
\choice{ we can't determine convergence or divergence from the Root Test  because the test is inconclusive}
\end{multipleChoice}


\end{exercise}

\end{exercise}
\end{exercise}
