\begin{exercise}
Suppose that $\{a_n\}_{n=1}$ is a sequence of positive terms and set $s_n = \sum_{k=1}^n a_k$.  Suppose it is known that:

\[
\lim_{n \to \infty} \sqrt[n]{a_n} = 3
\]


Select all of the following that \emph{must} be true:

\begin{selectAll}
\choice[correct]{$\sum_{k=1}^{\infty} a_k$ must diverge.}
\choice{$s_n$ could be bounded above.}
\choice[correct]{$\{s_n\}$ is monotonic.}
\choice[correct]{$\lim_{n \to \infty} s_n$ does not exist.}
\choice[correct]{$\sum_{k=1}^{\infty} s_k$ must diverge.}
\end{selectAll}

\begin{hint}
Since $a_n \geq 0$ for all $n$, $\{s_n\}$ :
\begin{multipleChoice}
\choice[correct]{must be increasing.}
\choice{must be decreasing.}
\choice{could be increasing or decreasing.}
\end{multipleChoice}

The Root Test guarantees that $\sum_{k=1}^{\infty} a_k$:

\begin{multipleChoice}
\choice{must converge.}
\choice[correct]{must diverge.}
\choice{could converge or diverge.}
\end{multipleChoice}

\begin{question}

Since $\sum_{k=1}^{\infty} a_k$ diverges, $\lim_{n \to \infty} s_n$ :
\begin{multipleChoice}
\choice{must exist.}
\choice[correct]{does not exist.}
\choice{could exist or not exist.}
\end{multipleChoice}
\begin{question}
Since $\lim_{n \to \infty} s_n$ does not exist:
\begin{multipleChoice}
\choice{$\sum_{k=1}^{\infty} s_k$ converges.}
\choice[correct]{$\sum_{k=1}^{\infty} s_k$ diverges by divergence test.}
\choice{$\sum_{k=1}^{\infty} s_k$ could converge or could diverge.}
\end{multipleChoice}

\end{question}
\end{question}

\end{hint}

\end{exercise}
