\begin{exercise}

Select all of the following series that converge.

\begin{selectAll}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{(-1)^k}{\sqrt[3]{k}}$}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{(-1)^k}{\sqrt[4]{k^5}}$}
\choice{$\sum_{k=1}^{\infty} \frac{(-5)^k}{3^k}$}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{k^2+1}{6^k}$}
\end{selectAll}

Select all of the following series that converge absolutely.

\begin{selectAll}
\choice{$\sum_{k=1}^{\infty} \frac{(-1)^k}{\sqrt[3]{k}}$}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{(-1)^k}{\sqrt[4]{k^5}}$}
\choice{$\sum_{k=1}^{\infty} \frac{(-5)^k}{3^k}$}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{k^2+1}{6^k}$}
\end{selectAll}

Select all of the following series that converge conditionally.

\begin{selectAll}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{(-1)^k}{\sqrt[3]{k}}$}
\choice{$\sum_{k=1}^{\infty} \frac{(-1)^k}{\sqrt[4]{k^5}}$}
\choice{$\sum_{k=1}^{\infty} \frac{(-5)^k}{3^k}$}
\choice{$\sum_{k=1}^{\infty} \frac{k^2+1}{6^k}$}
\end{selectAll}
\end{exercise}
