\outcome{Compute integrals involving powers of sine and cosine, Recognize the patterns that appear in trigonometric integrals and use appropriate substitutions 
to compute them.}


\begin{document}
\begin{exercise}
Determine the integral

\[
\int \sin^{4}(x) \d x= \answer{     \frac{3}{8}x-\frac{1}{4} \sin(2x) + \frac{1}{32} \sin(4x) +C} 
\]
(Use $C$ for the constant of integration)

\end{exercise}
