\begin{exercise}
Consider the series $\sum_{k=1}^{\infty} \left(\frac{2}{3}\right)^k$. Select all that apply:
\begin{multipleChoice}
\choice[correct]{This is a geometric series.  It converges.}
\choice{This is a geometric series.  It diverges.}
\choice{This is a $p$-series.  It converges.}
\choice{This is a $p$-series.  It diverges.}
\choice{This series diverges by the divergence test.}
\end{multipleChoice}
\end{exercise}
